import { ModelSelector } from '@tui-kit-ai/ai';

const modelSelector = new ModelSelector({
  showProvider: true,
  showCapabilities: true,
  showPerformance: true,
  showCost: true,
  showFilters: true,
  comparisonMode: false,
  onModelSelect: (model) => console.log('Selected:', model),
  onModelCompare: (models) => console.log('Comparing:', models)
});

// Set filters
modelSelector.setFilters({
  provider: 'openai',
  capabilities: ['text-generation', 'function-calling'],
  maxCost: 0.01
});

// Get selected model
const selected = modelSelector.getSelectedModel();
console.log('Selected model:', selected);
