## Create your first app

Let's build a simple AI-powered chat application. Create a new file called `index.ts`:

<CodeGroup>

```typescript index.ts
import { useTerminal, Box, TextInput, Text } from '@tui-kit-ai/core';
import { ChatContainer, AIService } from '@tui-kit-ai/ai';

// Initialize terminal
const { screen } = useTerminal();

// Setup AI service
const aiService = new AIService({
  provider: 'openai', // or 'anthropic', 'ollama'
  model: 'gpt-4',
  apiKey: process.env.OPENAI_API_KEY
});

// Create main container
const mainBox = new Box({
  parent: screen,
  top: 0,
  left: 0,
  right: 0,
  bottom: 3,
  border: { type: 'line' },
  label: ' AI Chat Assistant '
});

// Create chat container
const chat = new ChatContainer({
  parent: mainBox.el,
  top: 1,
  left: 1,
  right: 1,
  bottom: 1,
  scrollable: true,
  messages: [
    { role: 'assistant', content: 'Hello! I\'m your AI assistant. How can I help you today?' }
  ],
  onMessageSubmit: async (content) => {
    // Add user message to chat
    chat.addMessage({ role: 'user', content });
    
    // Stream AI response
    const stream = await aiService.streamCompletion([
      ...chat.getMessages(),
      { role: 'user', content }
    ]);
    
    // Handle streaming response
    let assistantMessage = '';
    for await (const chunk of stream.textStream) {
      assistantMessage += chunk;
      chat.updateLastMessage(assistantMessage);
      screen.render();
    }
  }
});

// Create input box
const inputBox = new TextInput({
  parent: screen,
  bottom: 0,
  left: 0,
  right: 0,
  height: 3,
  border: { type: 'line' },
  label: ' Type your message ',
  placeholder: 'Enter your message and press Enter...',
  onSubmit: (value) => {
    if (value.trim()) {
      chat.handleSubmit(value);
    }
  }
});

// Focus input and render
inputBox.focus();
screen.render();

// Handle exit
screen.key(['escape', 'q', 'C-c'], () => {
  process.exit(0);
});
```

## Environment Setup

Create a `.env` file in your project root:

```bash .env
# Choose your AI provider
OPENAI_API_KEY=your-openai-api-key
# ANTHROPIC_API_KEY=your-anthropic-api-key
# OLLAMA_HOST=http://localhost:11434

# Optional: Custom model settings
AI_MODEL=gpt-4
AI_TEMPERATURE=0.7
```

## Run your application

<CodeGroup>

```bash Development
# Using tsx for TypeScript
npx tsx index.ts

# Or add to package.json scripts
npm run dev
```

```bash Production
# Build first
npx tsc

# Then run
node dist/index.js
```

## Troubleshooting

<AccordionGroup>
  <Accordion title="Module not found errors">
    Make sure you have installed all required dependencies:
    ```bash
    npm install @tui-kit-ai/core @tui-kit-ai/ai @tui-kit-ai/agents
    ```
    
    For TypeScript projects, also install development dependencies:
    ```bash
    npm install --save-dev typescript @types/node tsx
    ```
  </Accordion>

  <Accordion title="Terminal rendering issues">
    Ensure your terminal supports ANSI colors and UTF-8 encoding:
    
    - **Windows**: Use Windows Terminal or WSL
    - **macOS**: Terminal.app or iTerm2 work great
    - **Linux**: Most modern terminal emulators are supported
    
    If you experience rendering issues, try:
    ```typescript
    import { useTerminal } from '@tui-kit-ai/core';
    
    const { screen } = useTerminal({
      smartCSR: true,
      fullUnicode: false // disable if you see strange characters
    });
    ```
  </Accordion>

  <Accordion title="AI API connection issues">
    Verify your API keys and endpoints:
    
    ```typescript
    // Test your API connection
    const aiService = new AIService({
      provider: 'openai',
      apiKey: process.env.OPENAI_API_KEY,
      // Add timeout and retry settings
      timeout: 30000,
      maxRetries: 3
    });
    
    // Test with a simple completion
    try {
      const response = await aiService.complete('Hello, world!');
      console.log('API connection successful');
    } catch (error) {
      console.error('API connection failed:', error);
    }
    ```
  </Accordion>
</AccordionGroup>
