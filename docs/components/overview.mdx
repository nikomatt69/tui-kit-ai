## Component Architecture

All TUI-Kit-AI components follow a consistent architecture pattern:

### Base Component Interface

```typescript
interface Component<T extends Widgets.Node> {
  el: T;                    // Underlying blessed.js element
  theme: ThemeConfig;       // Theme configuration
  destroy: () => void;      // Cleanup method
}

interface BaseProps {
  parent: Widgets.Node;     // Parent container (required)
  theme?: ThemeConfig;      // Optional theme override
  
  // Position & Size (blessed.js compatible)
  top?: string | number;
  left?: string | number;
  right?: string | number;
  bottom?: string | number;
  width?: string | number;
  height?: string | number;
  
  // Styling
  style?: StyleConfig;
  border?: BorderConfig;
  label?: string;
  
  // Behavior
  hidden?: boolean;
  clickable?: boolean;
  focusable?: boolean;
}
```

### Component Creation Pattern

Every component follows this pattern:

<CodeGroup>

// Add content

// Render

## Component Composition

Build complex interfaces by combining components:

```typescript
class ChatApplication {
  private screen: any;
  private chatContainer: ChatContainer;
  private inputSection: Box;
  private statusBar: StatusBar;

  constructor() {
    const { screen } = useTerminal();
    this.screen = screen;
    
    this.createLayout();
    this.setupEventHandlers();
  }

  private createLayout() {
    // Main chat area
    this.chatContainer = new ChatContainer({
      parent: this.screen,
      top: 0,
      left: 0,
      right: 0,
      bottom: 4,
      border: { type: 'line' },
      label: ' AI Chat '
    });

    // Input section
    this.inputSection = new Box({
      parent: this.screen,
      bottom: 2,
      left: 0,
      right: 0,
      height: 3,
      border: { type: 'line' },
      label: ' Message '
    });

    const input = new TextInput({
      parent: this.inputSection.el,
      top: 0,
      left: 1,
      right: 1,
      height: 1,
      placeholder: 'Type your message...',
      onSubmit: (value) => this.handleMessage(value)
    });

    // Status bar
    this.statusBar = new StatusBar({
      parent: this.screen,
      bottom: 0,
      left: 0,
      right: 0,
      height: 1,
      content: 'Ready to chat!'
    });
  }

  private async handleMessage(message: string) {
    // Add user message
    this.chatContainer.addMessage({
      role: 'user',
      content: message
    });

    // Show typing indicator
    this.statusBar.setContent('ü§ñ AI is thinking...');

    // Get AI response
    const response = await this.getAIResponse(message);
    
    // Add AI response
    this.chatContainer.addMessage({
      role: 'assistant',
      content: response
    });

    this.statusBar.setContent('Ready to chat!');
    this.screen.render();
  }
}
```

### Custom Themes

Create your own themes:

```typescript
import { createTheme } from '@tui-kit-ai/core';

const customTheme = createTheme({
  name: 'custom-ai-theme',
  colors: {
    primary: '#00ff88',      // Green accent
    secondary: '#0088ff',    // Blue accent  
    background: '#1a1a1a',   // Dark background
    foreground: '#ffffff',   // White text
    success: '#00ff00',      // Success green
    warning: '#ffaa00',      // Warning orange
    error: '#ff0000',        // Error red
    muted: '#666666'         // Muted gray
  },
  
  borders: {
    style: 'single',         // Border style
    focusStyle: 'double',    // Focused border
    color: 'primary',        // Default border color
    focusColor: 'secondary'  // Focused border color
  },
  
  typography: {
    heading: {
      style: { bold: true, fg: 'primary' }
    },
    body: {
      style: { fg: 'foreground' }
    },
    caption: {
      style: { fg: 'muted' }
    }
  }
});

// Apply theme globally

// Or apply to specific components

  parent: screen,
  theme: customTheme
});
```

## Best Practices

### Component Lifecycle

```typescript
class ComponentManager {
  private components: Set<any> = new Set();
  
  createComponent<T>(ComponentClass: new (...args: any[]) => T, props: any): T {
    const component = new ComponentClass(props);
    this.components.add(component);
    return component;
  }
  
  destroyComponent(component: any) {
    if (component.destroy) {
      component.destroy();
    }
    this.components.delete(component);
  }
  
  cleanup() {
    for (const component of this.components) {
      this.destroyComponent(component);
    }
  }
}

// Always cleanup on app exit
process.on('exit', () => {
  componentManager.cleanup();
});
```

### Performance Optimization

```typescript
class OptimizedApp {
  private renderScheduled = false;
  
  // Batch rendering updates
  scheduleRender() {
    if (!this.renderScheduled) {
      this.renderScheduled = true;
      setImmediate(() => {
        this.screen.render();
        this.renderScheduled = false;
      });
    }
  }
  
  // Efficient component updates
  updateComponents(updates: ComponentUpdate[]) {
    for (const update of updates) {
      update.component.update(update.data);
    }
    this.scheduleRender(); // Single render call
  }
}
```

### Error Handling

```typescript
class ComponentErrorBoundary {
  static wrap<T>(component: T, errorHandler: (error: Error) => void): T {
    return new Proxy(component, {
      get(target, prop) {
        const value = target[prop];
        
        if (typeof value === 'function') {
          return (...args: any[]) => {
            try {
              return value.apply(target, args);
            } catch (error) {
              errorHandler(error);
            }
          };
        }
        
        return value;
      }
    });
  }
}

// Usage
const safeComponent = ComponentErrorBoundary.wrap(
  new TextInput({ ... }),
  (error) => console.error('Component error:', error)
);
```
