## Box Component

The `Box` component is the most fundamental layout component, providing a basic container with optional borders, labels, and styling.

### Basic Usage

<CodeGroup>

```typescript Basic Box
import { useTerminal, Box } from '@tui-kit-ai/core';

const { screen } = useTerminal();

const basicBox = new Box({
  parent: screen,
  top: 2,
  left: 2,
  width: 30,
  height: 8,
  border: { type: 'line' },
  label: ' Basic Box '
});

screen.render();
```

## Flex Component

The `Flex` component provides flexible layout capabilities similar to CSS Flexbox, allowing you to create responsive layouts that adapt to content and container size.

### Basic Flexbox Layout

<CodeGroup>

```typescript Row Layout
import { Flex, Box } from '@tui-kit-ai/core';

const rowContainer = new Flex({
  parent: screen,
  top: 2,
  left: 2,
  right: 2,
  height: 10,
  direction: 'row',
  justify: 'space-between',
  align: 'stretch',
  gap: 1
});

// Add flex items
new Box({
  parent: rowContainer.el,
  flex: 1,
  border: { type: 'line' },
  label: ' Item 1 '
});

new Box({
  parent: rowContainer.el,
  flex: 2,  // Takes 2x space

  label: ' Item 2 (2x) '

new Box({
  parent: rowContainer.el,
  flex: 1,
  border: { type: 'line' },

```

## Grid Component

The `Grid` component provides a grid-based layout system for organizing content in rows and columns.

### Basic Grid Layout

<CodeGroup>

```typescript Simple Grid
import { Grid, Box } from '@tui-kit-ai/core';

const grid = new Grid({
  parent: screen,
  top: 2,
  left: 2,
  right: 2,
  bottom: 2,
  rows: 3,
  cols: 4,
  gap: 1
});

// Grid items are automatically positioned
for (let i = 0; i < 12; i++) {
  new Box({
    parent: grid.el,
    border: { type: 'line' },
    label: ` Cell ${i + 1} `
  });
}
```

```typescript Custom Grid Spans
const grid = new Grid({
  parent: screen,
  rows: 3,
  cols: 3,
  gap: 1
});

// Header spans full width
new Box({
  parent: grid.el,
  gridColumn: '1 / 4',  // Span all columns
  height: 5,
  border: { type: 'line' },
  label: ' Header '
});

// Content area
new Box({
  parent: grid.el,
  gridColumn: '1 / 3',  // Span 2 columns
  gridRow: '2 / 4',     // Span 2 rows
  border: { type: 'line' },
  label: ' Main Content '
});

// Sidebar
new Box({
  parent: grid.el,
  gridColumn: '3',
  gridRow: '2 / 4',
  border: { type: 'line' },
  label: ' Sidebar '
});
```

```typescript Responsive Grid
class ResponsiveGrid {
  private grid: Grid;
  private items: Box[] = [];
  
  constructor(parent: any) {
    this.grid = new Grid({
      parent,
      gap: 1
    });
    
    this.updateLayout();
    
    // Listen for resize events
    parent.on('resize', () => {
      this.updateLayout();
    });
  }
  
  private updateLayout() {
    const width = this.grid.el.width;
    const height = this.grid.el.height;
    
    // Responsive column calculation
    const cols = Math.max(1, Math.floor(width / 20));
    const rows = Math.max(1, Math.floor(height / 8));
    
    this.grid.setLayout(rows, cols);
  }
  
  addItem(content: string) {
    const item = new Box({
      parent: this.grid.el,
      border: { type: 'line' },
      label: ` ${content} `
    });
    
    this.items.push(item);
  }
}
```

```typescript Interactive Panel
class InteractivePanel {
  private panel: Panel;
  private content: string = '';
  
  constructor(parent: any) {
    this.panel = new Panel({
      parent,
      title: 'Interactive Panel',
      resizable: true,
      movable: true,
      closable: true
    });
    
    this.setupEventHandlers();
  }
  
  private setupEventHandlers() {
    // Close button
    this.panel.on('close', () => {
      this.panel.destroy();
    });
    
    // Resize handling
    this.panel.on('resize', (width, height) => {
      this.updateLayout(width, height);
    });
    
    // Key handlers
    this.panel.on('key', (ch, key) => {
      if (key.name === 'r') {
        this.refresh();
      }
    });
  }
  
  updateContent(newContent: string) {
    this.content = newContent;
    this.panel.setContent(this.content);
  }
  
  private refresh() {
    this.panel.setHeader(`Updated: ${new Date().toLocaleTimeString()}`);
    this.panel.render();
  }
}
```

```typescript Panel with Tabs
class TabbedPanel {
  private panel: Panel;
  private tabs: { [key: string]: string } = {};
  private currentTab: string = '';
  
  constructor(parent: any) {
    this.panel = new Panel({
      parent,
      title: 'Tabbed Interface',
      showTabs: true
    });
  }
  
  addTab(id: string, title: string, content: string) {
    this.tabs[id] = content;
    this.panel.addTab(id, title);
    
    if (!this.currentTab) {
      this.switchTab(id);
    }
  }
  
  switchTab(id: string) {
    if (this.tabs[id]) {
      this.currentTab = id;
      this.panel.setActiveTab(id);
      this.panel.setContent(this.tabs[id]);
    }
  }
}

// Usage
const tabbedPanel = new TabbedPanel(screen);
tabbedPanel.addTab('chat', 'Chat', 'Chat conversation content...');
tabbedPanel.addTab('settings', 'Settings', 'Configuration options...');
```

## Layout Best Practices

### 1. Responsive Design

```typescript
class ResponsiveLayout {
  private screen: any;
  private layout: 'mobile' | 'tablet' | 'desktop' = 'desktop';
  
  constructor(screen: any) {
    this.screen = screen;
    this.updateLayout();
    
    screen.on('resize', () => {
      this.updateLayout();
    });
  }
  
  private updateLayout() {
    const width = this.screen.width;
    
    // Determine layout breakpoints
    if (width < 60) {
      this.layout = 'mobile';
    } else if (width < 100) {
      this.layout = 'tablet';
    } else {
      this.layout = 'desktop';
    }
    
    this.rebuildLayout();
  }
  
  private rebuildLayout() {
    // Clear existing layout
    this.screen.children.forEach(child => child.destroy());
    
    switch (this.layout) {
      case 'mobile':
        this.createMobileLayout();
        break;
      case 'tablet':
        this.createTabletLayout();
        break;
      case 'desktop':
        this.createDesktopLayout();
        break;
    }
    
    this.screen.render();
  }
}
```

### 3. Memory Management

```typescript
class LayoutManager {
  private layouts: Map<string, any> = new Map();
  private currentLayout: string | null = null;
  
  registerLayout(id: string, layoutFactory: () => any) {
    this.layouts.set(id, layoutFactory);
  }
  
  switchLayout(id: string) {
    // Cleanup current layout
    if (this.currentLayout) {
      const current = this.layouts.get(this.currentLayout);
      if (current && current.destroy) {
        current.destroy();
      }
    }
    
    // Create new layout
    const layoutFactory = this.layouts.get(id);
    if (layoutFactory) {
      const newLayout = layoutFactory();
      this.currentLayout = id;
      return newLayout;
    }
  }
  
  cleanup() {
    this.layouts.forEach((layout) => {
      if (layout.destroy) {
        layout.destroy();
      }
    });
    this.layouts.clear();
  }
}
```
